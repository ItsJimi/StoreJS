    • store-data

    Finally, the simplest way to store data

    Install

    npm i store-data

    Example

    const Store = require('store-data')

const users = new Store({
  save: false
})

users.setSync('user123', {
  name: 'Jimi',
  dev: true
})

console.log(users.getSync('user123'))
// { name: 'Jimi', dev: true }

    or

    const Store = require('store-data')

const users = new Store({
  save: false
})

users.set('user123', {
  name: 'Jimi',
  dev: true
}).then(() => {
  users.get('user123').then(value => {
    console.log(value)
    // { name: 'Jimi', dev: true }
  })
})
